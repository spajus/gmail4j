33  /**
34   * Gmail RSS feed based {@link GmailClient} implementation.
35   * <p>
36   * It requires {@link HttpGmailConnection}. Functionality is limited to unread 
37   * message retrieval.
38   * <p>
39   * Example use:
40   * <p><blockquote><pre>
41   *     GmailClient client = new RssGmailClient();
42   *     client.setConnection(new HttpGmailConnection(gmailUser, gmailPass));
43   *     for (GmailMessage message : client.getUnreadMessages()) {
44   *         System.out.println(message.getFrom() + ": " + message.getTitle());
45   *     }
46   * </pre></blockquote>
47   * 
48   * @see RssGmailMessage
49   * @see GmailClient
50   * @see HttpGmailConnection
51   * @author Tomas Varaneckas &lt;tomas.varaneckas@gmail.com&gt;
52   * @version $Id: RssGmailClient.java 37 2009-07-09 17:26:41Z tomas.varaneckas $
53   * @since 0.1
54   */
55  public class RssGmailClient extends GmailClient {
56      
57      /**
58       * Gmail feed URL. It should remain as is, however you can set a new value.
59       * 
60       * @see #setGmailFeedUrl(String)
61       */
62      private String gmailFeedUrl = "https://mail.google.com/mail/feed/atom";
63  
64      /**
65       * Gets {@link #gmailFeedUrl}
66       * 
67       * @return Gmail feed URL
68       */
69      public String getGmailFeedUrl() {
70          return gmailFeedUrl;
71      }
72  
73      /**
74       * Sets {@link #gmailFeedUrl}
75       * <p>
76       * Use it only if you would like to use another Gmail feed URL than the 
77       * default <a href="https://mail.google.com/mail/feed/atom">
78       * https://mail.google.com/mail/feed/atom</a>.
79       * 
80       * @param gmailFeedUrl New Gmail feed URL
81       * @throws GmailException if something goes wrong
82       */
83      public void setGmailFeedUrl(final String gmailFeedUrl) {
84          this.gmailFeedUrl = gmailFeedUrl;
85      }
86      
87      @Override
88      public List<GmailMessage> getUnreadMessages() {
89          final List<GmailMessage> messages = new ArrayList<GmailMessage>();
90          try {
91              //for ROME properties loader
92              Thread.currentThread().setContextClassLoader(getClass()
93                      .getClassLoader());
94              HttpGmailConnection c = getGmailConnection();
95              c.setUrl(gmailFeedUrl);
96              final URLConnection con = c.openConnection();
97              final SyndFeedInput feedInput = new SyndFeedInput();
98              final SyndFeed gmail = feedInput.build(new XmlReader(con));
99              for (final Object entry : gmail.getEntries()) {
100                 if (entry instanceof SyndEntry) {
101                     messages.add(new RssGmailMessage((SyndEntry) entry));
102                 }
103             }
104             if (log.isDebugEnabled()) {
105                 log.debug("Got " + messages.size() + " new messages.");
106             }
107         } catch (final Exception e) {
108             throw new GmailException("Failed getting unread messages", e);
109         }
110         return messages;
111     }
112     
113     /**
114      * Gets {@link HttpGmailConnection} (casts it from parent 
115      * {@link GmailConnection})
116      * 
117      * @return HttpGmailConnection if available
118      * @throws GmailException if connection is not an instance of 
119      *         HttpGmailConnection
120      */
121     private HttpGmailConnection getGmailConnection() {
122         if (connection instanceof HttpGmailConnection) {
123             return (HttpGmailConnection) connection;
124         } 
125         throw new GmailException("RssGmailClient requires HttpGmailConnection!");
126     }
127 
128     @Override
129     public void send(final GmailMessage message) {
130         throw new UnsupportedOperationException("RssGmailClient is not " +
131         		"capable of sending messages.");
132     }
133     
134 }
