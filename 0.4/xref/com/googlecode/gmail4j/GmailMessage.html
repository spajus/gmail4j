24  /**
25   * Gmail message interface. 
26   * <p>
27   * Implementations that do not provide some functionality (i.e. 
28   * {@link #getContentText()}) throw {@link UnsupportedOperationException}
29   * for those methods by default. 
30   * <p>
31   * Example: Print unread messages:
32   * <p><blockquote><pre>
33   *     GmailClient client = //...instantate
34   *     //configure the client...
35   *     for (GmailMessage message : client.getUnreadMessages()) {
36   *         System.out.println(message.getFrom());
37   *         System.out.println(message.getDate());
38   *         System.out.println(message.getTitle());
39   *         System.out.println(message.getPreview());
40   *         System.out.println(message.getLink());
41   *         System.out.println(message.getContentText());
42   *     }
43   * </pre></blockquote><p>
44   * Example: Send a simple message:
45   * <p><blockquote><pre>
46   *     GmailClient client = //...instantate
47   *     //configure the client...
48   *     GmailMessage message = //new GmailMessage instance
49   *     message.setSubject("Subject");
50   *     message.setContentText("Content");
51   *     message.addTo(new EmailAddress("j.smith@example.com");
52   *     client.send(message);
53   * </pre></blockquote></p>
54   * 
55   * @see GmailMessage
56   * @author Tomas Varaneckas &lt;tomas.varaneckas@gmail.com&gt;
