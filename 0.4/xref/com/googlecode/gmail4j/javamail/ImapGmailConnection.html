40  /**
41   * JavaMail based IMAP {@link GmailConnection} implementation.
42   * <p>
43   * ImapGmailConnection is {@link ProxyAware}, allowing to use JavaMail IMAP
44   * via HTTP proxy, which was not possible by JavaMail design.
45   * <p>
46   * Example:
47   * <p><blockquote><pre>
48   *     GmailConnection conn = new ImapGmailConnection();
49   *     conn.setLoginCredentials(new Credentials("user", "pass".toCharArray()));
50   *     //if proxy is required
51   *     ((ProxyAware) conn).setProxy("proxy.example.com", 8080);
52   *     //if proxy auth is required
53   *     ((ProxyAware) conn).setProxyCredentials("proxyuser", 
54   *         "proxypass".toCharArray());
55   *     GmailClient client = new ImapGmailClient();
56   *     client.setConnection(conn);
57   * </pre></blockquote></p>
58   * 
59   * @author Tomas Varaneckas &lt;tomas.varaneckas@gmail.com&gt;
60   * @version $Id: ImapGmailConnection.java 44 2010-10-25 10:36:43Z rajivderas@gmail.com $
61   * @since 0.3
62   */

245     /**
246      * Opens Gmail {@link Store}
247      * 
248      * @return singleton instance of Gmail {@link Store}
249      */
250     public Store openGmailStore() {
251         try {
252             // if stort object instance is not null and the service connection state
253             // is connected then close this service and terminate its connection.
254             // Note : due to security concerns gmail imap only allow max 10 
255             // connections because of this reason any existing connection that 
256             // is open should terminate.
257             if (isConnected()) {
258                 disconnect();
259             }            
260             store = getSession().getStore("imaps");
261             store.addConnectionListener(new ImapConnectionHandler(
262                     new ConnectionInfo(loginCredentials.getUsername(),
263                     gmailImapHost,
264                     gmailImapPort)));
265             store.connect(gmailImapHost, loginCredentials.getUsername()
266                     .concat(CommonConstants.GMAIL_EXTENTION),
267                     new String(loginCredentials.getPasword()));
268             setConnected(store.isConnected());
269         } catch (final Exception e) {
270             throw new GmailException("Failed opening Gmail IMAP store", e);
271         }
272         return store;
273     }

275     /**
276      * Gets Gmail {@link Transport}
277      * 
278      * @return Configured and ready for use Transport
279      */
280     public Transport getTransport() {
281         try {
282             final Transport transport = getSession().getTransport();
283             transport.addConnectionListener(new ImapConnectionHandler(
284                     new ConnectionInfo(loginCredentials.getUsername(),
285                     gmailSmtpHost,
286                     gmailSmtpPort)));
287             transport.connect(loginCredentials.getUsername(),
288                     new String(loginCredentials.getPasword()));
289             return transport;
290         } catch (final Exception e) {
291             throw new GmailException("ImapGmailClient requires ImapGmailConnection!");
292         }
293     }

295     /**
296      * Gets Gmail {@link Session}
297      * 
298      * @return Configured and ready for use Gmail {@link Session}
299      */
300     public Session getSession() {
301         synchronized (this) {
302             if (mailSession == null) {
303                 properties = System.getProperties();
304                 properties.put("mail.transport.protocol", "smtps");
305                 properties.put("mail.debug", true);
306                 properties.put("mail.smtps.host", gmailSmtpHost);
307                 properties.put("mail.smtps.port", gmailSmtpPort);
308                 if (proxy != null) {
309                     log.debug("Setting proxy: " + proxy.address());
310                     final SocketFactory sf =
311                             new HttpProxyAwareSslSocketFactory(proxy, proxyCredentials);
312                     properties.put("mail.imap.host", gmailImapHost);
313                     properties.put("mail.imaps.ssl.socketFactory", sf);
314                     properties.put("mail.imaps.ssl.socketFactory.fallback", "false");
315                     properties.put("mail.imaps.ssl.socketFactory.port", gmailImapPort);
316                 }
317                 if (proxyCredentials != null) {
318                     mailSession = Session.getInstance(properties);
319                 } else {
320                     mailSession = Session.getInstance(properties);
321                 }
322                 for (Provider p : mailSession.getProviders()) {
323                     log.debug(p);
324                 }
325             }
326         }
327         return mailSession;
328     }

333     public void disconnect() {
334         try {
335             if (isConnected()) {

339             log.warn("Failed closing Gmail IMAP store", e);
