17  package com.googlecode.gmail4j.javamail;
18  
19  import javax.mail.event.ConnectionAdapter;
20  import javax.mail.event.ConnectionEvent;
21  import org.apache.commons.logging.Log;
22  import org.apache.commons.logging.LogFactory;
23  
24  /**
25   * JavaMail based IMAP {@link ConnectionAdapter} for Connection events.
26   * <p>
27   * ImapConnectionHandler is the adapter which receives connection events.
28   * This class logs the connection event services if
29   * {@code getSession().setDebug(true)}.
30   * <p>
31   * Example:
32   * <p><blockquote><pre>
33   *     // if implementing a {@link Transport protcol}.
34   *     Transport transport = getSession().getTransport();
35   *     transport.addConnectionListener(new ImapConnectionHandler(
36   *                  new ConnectionInfo(loginCredentials.getUsername(),
37   *                  gmailSmtpHost,
38   *                  gmailSmtpPort)));
39   * </pre></blockquote></p>
40   *
41   * @author Rajiv Perera &lt;rajivderas@gmail.com&gt;
42   * @version $Id: ImapConnectionHandler.java 42 2010-10-23 ‏‎15:04:41Z rajivderas@gmail.com $
43   * @since 0.4
44   */
45  public class ImapConnectionHandler extends ConnectionAdapter {
46  
47      /**
48      * Logger
49      */
50      private static final Log LOG = LogFactory.getLog(ImapConnectionHandler.class);
51      /**
52       * Contain information of current connection service{@link ConnectionInfo}.
53       *
54       * @see #ConnectionInfo(String user, String host, int port)
55       */
56      private final ConnectionInfo connectionInfo;
57      /**
58       * Holds the current connection service start time in milli seconds.
59       */
60      private long startTime;
61  
62      /**
63       * Constructor with ConnectionInfo object
64       *
65       * @param connectionInfo ConnectionInfo object
66       */
67      public ImapConnectionHandler(ConnectionInfo connectionInfo) {
68          this.connectionInfo = connectionInfo;
69      }
70  
71      @Override
72      public void opened(ConnectionEvent e) {
73          if (LOG.isDebugEnabled()) {
74              startTime = System.currentTimeMillis();
75              LOG.debug("ImapGmailConnection " + connectionInfo + "is opened.");
76          }
77      }
78  
79      @Override
80      public void closed(ConnectionEvent e) {
81          if (LOG.isDebugEnabled()) {
82              long upTime = (System.currentTimeMillis() - startTime) / 1000;
83              LOG.debug("ImapGmailConnection " + connectionInfo 
84                      + "is closed after " + upTime + " seconds.");
85          }
86      }
87  
88      @Override
89      public void disconnected(ConnectionEvent e) {
90          if (LOG.isDebugEnabled()) {
91              long upTime = (System.currentTimeMillis() - startTime) / 1000;
92              LOG.debug("ImapGmailConnection " + connectionInfo 
93                      + "has been disconnectedafter " + upTime + " seconds.");
94          }
95      }
96  }
