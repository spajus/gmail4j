19  import java.io.Serializable;

21  /**
22   * Connection event information object for {@link ImapConnectionHandler}.
23   * <p>
24   * ConnectionInfo object contains information related
25   * to the current {@link Connection} service.
26   * <p>
27   *
28   * @author Rajiv Perera &lt;rajivderas@gmail.com&gt;

30   * @since 0.4
31   */
32  public class ConnectionInfo implements Serializable {
33  
34      private static final long serialVersionUID = 1L;
35  
36      /**
37       * Gmail Username for mail authentication.
38       */
39      private final String user;
40      /**
41       * Gmail SMTP/IMAP host for mail sending/receiving.
42       */
43      private final String host;
44      /**
45       * Gmail SMTP/IMAP port for mail sending/receiving.
46       */
47      private final int port;
48  
49      /**
50       * Constructor with Gmail user, host and port.
51       * 
52       * @param user Gmail username
53       * @param host Gmail SMTP/IMAP host for mail sending/receiving
54       * @param port Gmail SMTP/IMAP port for mail sending/receiving
55       */
56      public ConnectionInfo(final String user, final String host, final int port) {
57          this.user = user;
58          this.host = host;
59          this.port = port;
60      }
61  
62      /**
63       * Gets the {@link #user}
64       *
65       * @return username
66       */
67      public String getUser() {
68          return user;
69      }
70  
71      /**
72       * Gets the {@link #host}
73       *
74       * @return SMTP/IMAP host for mail sending/receiving
75       */
76      public String getHost() {
77          return host;
78      }
79  
80      /**
81       * Gets the {@link #port}
82       *
83       * @return SMTP/IMAP port for mail sending/receiving
84       */
85      public int getPort() {
86          return port;
87      }
88  
89      @Override
90      public String toString() {
91          StringBuilder toString = new StringBuilder();
92          try {
93              toString.append("ConnectionInfo:{user:").append(getUser())
94                      .append(";host:").append(getHost())
95                      .append(";port:").append(getPort())
96                      .append("} ");
97              return toString.toString();
98          } catch (final Exception e) {
99              toString = null;
100             return super.toString().concat("(e:")
101                     .concat(e.getMessage()).concat(")");
102         }
103     }

105     @Override
106     public boolean equals(final Object obj) {
107         if (obj == null) {
108             return false;
109         }
110         if (getClass() != obj.getClass()) {
111             return false;
112         }
113         final ConnectionInfo other = (ConnectionInfo) obj;
114         if ((this.user == null) ? (other.user != null)
115                 : !this.user.equals(other.user)) {
116             return false;
117         }
118         if ((this.host == null) ? (other.host != null)
119                 : !this.host.equals(other.host)) {
120             return false;
121         }
122         if (this.port != other.port) {
123             return false;
124         }
125         return true;
126     }
127 
128     @Override
129     public int hashCode() {
130         int hash = 7;
131         hash = 37 * hash + (this.user != null ? this.user.hashCode() : 0);
132         hash = 37 * hash + (this.host != null ? this.host.hashCode() : 0);
133         hash = 37 * hash + this.port;
134         return hash;
135     }
